import express from 'express';
import fetch from 'node-fetch';

const app = express();
const PORT = process.env.PORT || 3000;

app.use(express.static('public'));

// Fetch bootstrap static data
app.get('/api/bootstrap-static', async (req, res) => {
  try {
    const url = 'https://fantasy.premierleague.com/api/bootstrap-static/';
    const response = await fetch(url);
    if (!response.ok) {
      throw new Error('Failed to fetch bootstrap static data');
    }
    const data = await response.json();
    res.json(data);
  } catch (error) {
    console.error('Error fetching bootstrap static data:', error);
    res.status(500).json({ error: error.message });
  }
});

// Fetch specific gameweek live data
app.get('/api/event/:gameweek/live', async (req, res) => {
  const { gameweek } = req.params;
  try {
    const url = `https://fantasy.premierleague.com/api/event/${gameweek}/live/`;
    const response = await fetch(url);
    if (!response.ok) {
      throw new Error(`Failed to fetch live data for gameweek ${gameweek}`);
    }
    const data = await response.json();
    res.json(data);
  } catch (error) {
    console.error(`Error fetching live data for gameweek ${gameweek}:`, error);
    res.status(500).json({ error: error.message });
  }
});

app.listen(PORT, () => {
  console.log(`Server running on port ${PORT}`);
});
